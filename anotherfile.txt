                                <div class="song-meta">
                                    <span>${song.artist}</span>
                                    <i class="fas fa-circle"></i>
                                    <span>${song.file}</span>
                                </div>
                            </div>
                            <i class="fas fa-play" style="color: var(--primary);"></i>
                        </div>
                    `).join('')}
                </div>
            `;
            
            currentPlaylist = songsToDisplay;
        }

        // Setup audio listeners
        function setupAudioListeners() {
            audio.addEventListener('timeupdate', updateProgress);
            audio.addEventListener('ended', playNext);
            audio.addEventListener('loadedmetadata', () => {
                document.getElementById('totalTime').textContent = formatTime(audio.duration);
            });
            audio.addEventListener('error', (e) => {
                console.error('Audio error:', e);
                alert('❌ Cannot play: ' + currentPlaylist[currentIndex]?.file + '\nMake sure the file exists!');
            });
        }

        // Play song
        function playSong(index) {
            if (currentPlaylist.length === 0) return;
            
            currentIndex = index;
            const song = currentPlaylist[index];
            
            // Show player
            document.getElementById('player').classList.add('active');
            document.body.classList.add('player-active');
            
            // Update UI
            document.getElementById('currentSong').textContent = song.name;
            document.getElementById('currentArtist').textContent = song.artist;
            document.getElementById('playIcon').className = 'fas fa-pause';
            
            // Load and play
            audio.src = song.url;
            audio.load();
            audio.play().catch(e => {
                console.log('Playback error:', e);
            });
            
            isPlaying = true;
            
            // Update active state
            displaySongs(currentPlaylist, document.getElementById('songsTitle').innerText);
        }

        // Toggle play/pause
        function togglePlay() {
            if (currentIndex === -1) {
                if (currentPlaylist.length > 0) playSong(0);
                return;
            }
            
            if (isPlaying) {
                audio.pause();
                document.getElementById('playIcon').className = 'fas fa-play';
            } else {
                audio.play();
                document.getElementById('playIcon').className = 'fas fa-pause';
            }
            isPlaying = !isPlaying;
        }

        // Play next
        function playNext() {
            if (currentPlaylist.length === 0 || currentIndex === -1) return;
            let nextIndex = (currentIndex + 1) % currentPlaylist.length;
            playSong(nextIndex);
        }

        // Play previous
        function playPrevious() {
            if (currentPlaylist.length === 0 || currentIndex === -1) return;
            let prevIndex = (currentIndex - 1 + currentPlaylist.length) % currentPlaylist.length;
            playSong(prevIndex);
        }

        // Play remix (random from all)
        function playRemix() {
            if (songs.length === 0) return;
            const randomIndex = Math.floor(Math.random() * songs.length);
            currentPlaylist = songs;
            playSong(randomIndex);
            goBack();
        }

        // Update progress
        function updateProgress() {
            if (!isNaN(audio.duration)) {
                const progress = (audio.currentTime / audio.duration) * 100;
                document.getElementById('progressFill').style.width = progress + '%';
                document.getElementById('currentTime').textContent = formatTime(audio.currentTime);
            }
        }

        // Format time
        function formatTime(seconds) {
            if (isNaN(seconds)) return '0:00';
            const mins = Math.floor(seconds / 60);
            const secs = Math.floor(seconds % 60);
            return `${mins}:${secs.toString().padStart(2, '0')}`;
        }

        // Seek
        function seek(e) {
            const bar = e.currentTarget;
            const rect = bar.getBoundingClientRect();
            const percent = (e.clientX - rect.left) / rect.width;
            audio.currentTime = percent * audio.duration;
        }

        // Multi-select functions
        function openArtistModal() {
            if (artists.length < 2) return;
            
            const grid = document.getElementById('artistSelectGrid');
            grid.innerHTML = artists.map(artist => `
                <div style="background: var(--surface-light); border-radius: 16px; padding: 16px; text-align: center; cursor: pointer; border: 2px solid transparent;" 
                     onclick="toggleArtistSelect('${artist.name}', this)">
                    <div style="width: 60px; height: 60px; border-radius: 50%; background: var(--gradient); margin: 0 auto 10px; display: flex; align-items: center; justify-content: center; font-size: 24px; color: white;">
                        ${artist.name.charAt(0)}
                    </div>
                    <div style="font-weight: 600; margin-bottom: 4px;">${artist.name}</div>
                    <div style="color: var(--text-secondary); font-size: 12px;">${artist.songCount} songs</div>
                </div>
            `).join('');
            
            document.getElementById('artistModal').style.display = 'flex';
            selectedArtists = [];
            document.getElementById('modalSelectedCount').textContent = '0';
        }

        function toggleArtistSelect(artistName, element) {
            const index = selectedArtists.indexOf(artistName);
            if (index === -1) {
                selectedArtists.push(artistName);
                element.style.borderColor = 'var(--primary)';
            } else {
                selectedArtists.splice(index, 1);
                element.style.borderColor = 'transparent';
            }
            
            document.getElementById('modalSelectedCount').textContent = selectedArtists.length;
            document.getElementById('modalShuffleBtn').style.opacity = selectedArtists.length < 2 ? '0.5' : '1';
        }

        function closeArtistModal() {
            document.getElementById('artistModal').style.display = 'none';
        }

        function shuffleSelectedArtists() {
            if (selectedArtists.length < 2) return;
            
            const artistSongs = songs.filter(song => selectedArtists.includes(song.artist));
            if (artistSongs.length === 0) return;
            
            const randomIndex = Math.floor(Math.random() * artistSongs.length);
            currentPlaylist = artistSongs;
            playSong(randomIndex);
            goBack();
            closeArtistModal();
        }
    </script>
